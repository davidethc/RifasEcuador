# Token de autenticación de Payphone
# Obtén este token desde tu panel de Payphone (https://payphone.app)
# Para desarrollo: usa el token de SANDBOX
# Para producción: usa el token de PRODUCCIÓN
NEXT_PUBLIC_PAYPHONE_TOKEN=tu_token_payphone_aqui

# Store ID de tu tienda en Payphone
# Este ID identifica tu comercio en Payphone
NEXT_PUBLIC_PAYPHONE_STORE_ID=tu_store_id_aqui

# URL de tu aplicación
# Desarrollo local: http://localhost:3000
# Producción: https://tu-dominio.com
NEXT_PUBLIC_APP_URL=http://localhost:3000

# Ambiente de Payphone
# Valores posibles: 'sandbox' | 'production'
# - sandbox: Para pruebas y desarrollo
# - production: Para producción real
NEXT_PUBLIC_PAYPHONE_ENVIRONMENT=sandbox

# ============================================
# Configuración de Resend (Envío de correos)
# ============================================
#
# API Key de Resend para enviar correos de confirmación
# Obtén tu API Key desde: https://resend.com/api-keys
# Formato: re_xxxxxxxxxxxxx
RESEND_API_KEY=tu_resend_api_key_aqui

# Email desde el cual se enviarán los correos (opcional)
# Ejemplo: facturas / confirmación desde administracion@altokeec.com
# Formato: Nombre <email@dominio.com>
# IMPORTANTE: El dominio DEBE estar verificado en Resend
# RESEND_FROM_EMAIL=ALTOKEE <administracion@altokeec.com>

# ============================================
# Otras variables de entorno (si las necesitas)

# URL de tu proyecto Supabase
NEXT_PUBLIC_SUPABASE_URL=tu_url_supabase

# Anon Key (pública, segura para el cliente)
NEXT_PUBLIC_SUPABASE_ANON_KEY=tu_anon_key
#
# Service Role Key (PRIVADA - solo para operaciones del servidor)
# Obténla desde: Supabase Dashboard > Settings > API > service_role key
# IMPORTANTE: NUNCA la subas a GitHub, solo úsala en el servidor
# SUPABASE_SERVICE_ROLE_KEY=tu_service_role_key_aqui

# URL del GIF desde Vercel Blob Storage o Supabase Storage (opcional)
