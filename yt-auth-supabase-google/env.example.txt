# Token de autenticación de Payphone
# Obtén este token desde tu panel de Payphone (https://payphone.app)
# Para desarrollo: usa el token de SANDBOX
# Para producción: usa el token de PRODUCCIÓN
NEXT_PUBLIC_PAYPHONE_TOKEN=tu_token_payphone_aqui

# Store ID de tu tienda en Payphone
# Este ID identifica tu comercio en Payphone
NEXT_PUBLIC_PAYPHONE_STORE_ID=tu_store_id_aqui

# URL de tu aplicación
# Desarrollo local: http://localhost:3000
# Producción: https://tu-dominio.com
NEXT_PUBLIC_APP_URL=http://localhost:3000

# Ambiente de Payphone
# Valores posibles: 'sandbox' | 'production'
# - sandbox: Para pruebas y desarrollo
# - production: Para producción real
NEXT_PUBLIC_PAYPHONE_ENVIRONMENT=sandbox
